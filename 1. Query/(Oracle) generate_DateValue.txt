/* 날짜 생성 : 특정 일자 */
SELECT TO_CHAR(TO_DATE('2020-06-01') + LEVEL - 1, 'YYYY-MM-DD') AS GEN_DATE
FROM DUAL
CONNECT BY LEVEL <= (TO_DATE('2020-06-30') - TO_DATE('2020-06-01') + 1)
;

/* 날짜 생성 : 해당 월 */
SELECT TO_CHAR(TO_DATE(TO_CHAR(TRUNC(SYSDATE,'MM'),'YY/MM/DD')) + LEVEL - 1, 'YYYY-MM-DD') AS GEN_DATE
FROM DUAL
CONNECT BY LEVEL <= (LAST_DAY(SYSDATE) - TO_DATE(TO_CHAR(TRUNC(SYSDATE,'MM'),'YY/MM/DD')) + 1)
;
